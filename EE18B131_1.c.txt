#include <stdio.h>
int main()
{
    int n1, n2, i, gcd,a;
    printf("Enter two numbers seperated by a space:	");
    scanf("%d %d", &n1, &n2);
    for(i=1; i <= n1 && i <= n2; ++i)
    {
        // Checks if i is factor of both integers
        if(n1%i==0 && n2%i==0)
            gcd = i;
    }
	a=(n1+n2)/gcd;
	
    printf("%d",a);
    return 0;
}
